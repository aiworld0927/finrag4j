### 3.2 测试场景

#### 场景1：登录并发测试

```javascript
// k6 脚本：login-load-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '2m', target: 50 },   // 逐步增加到50并发
    { duration: '5m', target: 100 },  // 保持100并发
    { duration: '2m', target: 200 },  // 峰值200并发
    { duration: '2m', target: 0 },    // 逐步降载
  ],
  thresholds: {
    http_req_duration: ['p(95)<200'],
    http_req_failed: ['rate<0.1'],
  },
};

export default function () {
  const res = http.post('http://gateway:8085/api/auth/login', {
    username: 'test_user',
    password: 'test_password',
  });
  
  check(res, {
    'status is 200': (r) => r.status === 200,
    'response time < 200ms': (r) => r.timings.duration < 200,
  });
  
  sleep(1);
}
```

#### 场景2：检索性能测试

```javascript
// k6 脚本：search-load-test.js
export const options = {
  stages: [
    { duration: '1m', target: 20 },
    { duration: '3m', target: 50 },
    { duration: '3m', target: 100 },
    { duration: '1m', target: 0 },
  ],
  thresholds: {
    http_req_duration: ['p(95)<300'],
  },
};

export default function () {
  const res = http.post('http://gateway:8085/api/search', {
    query: '金融监管要求',
    kbId: 1,
    topK: 10,
  });
  
  check(res, {
    'status is 200': (r) => r.status === 200,
    'response time < 300ms': (r) => r.timings.duration < 300,
    'results returned': (r) => r.json('data.length') > 0,
  });
}
```

## 5. 性能优化建议

### 5.1 JVM 调优

```bash
# Gateway/Auth（轻量服务）
-Xms512m -Xmx1g -XX:+UseG1GC

# Search/Agent（计算密集型）
-Xms4g -Xmx8g -XX:+UseG1GC -XX:MaxGCPauseMillis=100

# Document（IO密集型）
-Xms2g -Xmx4g -XX:+UseG1GC -XX:+AlwaysPreTouch
```

### 5.2 数据库优化

```sql
-- 向量索引优化
CREATE INDEX ON vector_chunk 
USING ivfflat (embedding vector_cosine_ops) 
WITH (lists = 100);

-- 连接池配置
max_connections = 200
shared_buffers = 8GB
effective_cache_size = 24GB
work_mem = 64MB
```

### 5.3 缓存策略

| 数据 | 缓存方式 | TTL | 说明 |
|------|---------|-----|------|
| 用户信息 | Redis Hash | 30min | 登录态缓存 |
| 知识库元数据 | Redis String | 10min | 频繁读取 |
| 检索结果 | Redis ZSet | 5min | 热点查询 |
| 向量索引 | 本地内存 | - | 应用启动加载 |
